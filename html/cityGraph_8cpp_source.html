224void CityGraph::linkPoints(const point &p, const point &n, int direction, bool subPoints) {
225 std::vector<sf::Angle> anglesPoint = {p.angle, p.angle + sf::radians(M_PI)};
226 std::vector<sf::Angle> anglesNeighbor = {n.angle, n.angle + sf::radians(M_PI)};
227
228 point copyPoint = p;
229 point copyNeighbor = n;
230
231 bool isRiP = direction == 2 || direction == 0;
232 bool isRiN = direction == 2 || direction == 1;
233 bool isStraight = direction != 2;
234 isStraight &= (anglesPoint[0] == anglesNeighbor[0] || anglesPoint[0] == anglesNeighbor[1] ||
235 anglesPoint[1] == anglesNeighbor[0] || anglesPoint[1] == anglesNeighbor[1]);
236 isStraight &= subPoints;
237
238 if (!isStraight) {
239 for (const auto &anglePoint : anglesPoint) {
240 for (const auto &angleNeighbor : anglesNeighbor) {
241 copyPoint.angle = anglePoint;
242 copyNeighbor.angle = angleNeighbor;
243
244 neighbors[copyPoint].push_back({copyNeighbor, 0, 0, isRiP}); // This fields will be updated later
245 neighbors[copyNeighbor].push_back({copyPoint, 0, 0, isRiN});
246
247 graphPoints.insert(copyPoint);
248 graphPoints.insert(copyNeighbor);
249 }
250 }
251 return;
252 }
253
254 // Link adding points in the middle
255 double pointDistance = 3;
256 double distance = std::sqrt(std::pow(n.position.x - p.position.x, 2) + std::pow(n.position.y - p.position.y, 2));
257 int numPoints = distance / pointDistance;
258 double dx = (n.position.x - p.position.x) / numPoints;
259 double dy = (n.position.y - p.position.y) / numPoints;
260
261 for (const auto &anglePoint : anglesPoint) {
262 for (const auto &angleNeighbor : anglesNeighbor) {
263 point previousPoint = p;
264 previousPoint.angle = anglePoint;
265
266 for (int i = 1; i <= numPoints; i++) {
267 point newPoint;
268 newPoint.position = sf::Vector2f(p.position.x + i * dx, p.position.y + i * dy);
269 newPoint.angle = anglePoint;
270
271 neighbors[previousPoint].push_back({newPoint, 0, 0, isRiP}); // This fields will be updated later
272 neighbors[newPoint].push_back({previousPoint, 0, 0, isRiN});
273
274 previousPoint = newPoint;
275
276 graphPoints.insert(newPoint);
277 }
278
279 // Add the last point
280 neighbors[previousPoint].push_back({n, 0, 0, isRiP}); // This fields will be updated later
281 }
282 }

303bool CityGraph::canLink(const point &point1, const point &point2, double speed, double *distance) const {
304 double radius = turningRadius(speed);
305
306 ob::DubinsStateSpace space(radius, true);
307
308 ob::State *start = space.allocState();
309 ob::State *end = space.allocState();
310
311 start->as<ob::DubinsStateSpace::StateType>()->setXY(point1.position.x, point1.position.y);
312 start->as<ob::DubinsStateSpace::StateType>()->setYaw(point1.angle.asRadians());
313
314 end->as<ob::DubinsStateSpace::StateType>()->setXY(point2.position.x, point2.position.y);
315 end->as<ob::DubinsStateSpace::StateType>()->setYaw(point2.angle.asRadians());
316
317 double total = 0;
318
319 // Extract the path
320 ob::DubinsStateSpace::DubinsPath path = space.dubins(start, end);
321 for (unsigned int i = 0; i < 3; ++i) // Dubins path has up to 3 segments
322 {
323 auto type = (*path.type_)[i];
324 if (type == ob::DubinsStateSpace::DubinsPathSegmentType::DUBINS_LEFT) {
325 total += std::abs(path.length_[i]);
326 } else if (type == ob::DubinsStateSpace::DubinsPathSegmentType::DUBINS_RIGHT) {
327 total += std::abs(path.length_[i]);
328 }
329 }
330
331 *distance = space.distance(start, end);
332 return total < M_PI * 0.75f;
333}
