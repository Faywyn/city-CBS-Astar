73bool ManagerOCBS::findConflict(int *car1, int *car2, int *time, Node *node) {
74 int maxPathLength = -1;
75 for (int i = 0; i < numCars; i++) {
76 maxPathLength = std::max(maxPathLength, (int)node->paths[i].size());
77 }
78
79 for (int t = 0; t < maxPathLength; t++) {
80 for (int i = 0; i < numCars; i++) {
81 for (int j = i + 1; j < numCars; j++) {
82 sf::Vector2f diff = node->paths[i][t] - node->paths[j][t];
83 double len = std::sqrt(diff.x * diff.x + diff.y * diff.y);
84 if (len < CAR_LENGTH * COLLISION_SAFETY_FACTOR) {
85 *car1 = i;
86 *car2 = j;
87 *time = t;
88 return true;
89 }
90 }
91 }
92 }
93
94 return false;

98 if (openSet.empty()) {
99 spdlog::info("No solution found");

102

104 openSet.pop();
105
106 spdlog::debug("Processing node with cost: {}", node.cost);
107
108 int car1, car2, time;
109 if (!findConflict(&car1, &car2, &time, &node)) {

111

115

117 }
118
119 spdlog::debug("Found conflict between car {} and car {} at time {}", car1, car2, time);

122 int car1Index = car1;
123 int car2Index = car2;
124
125 double ratio1 = node.costs[car1] / baseCosts[car1];
126 double ratio2 = node.costs[car2] / baseCosts[car2];

129 car1Index = car2;
130 car2Index = car1;
131 }
132
133 ConflictSituation situation1;
134 situation1.car = car1Index;

136 situation1.at = node.paths[car1Index][time];

139 situation2.time = time * SIM_STEP_TIME;

144 conflict1.withCar = car2Index;
145 conflict1.time = time * SIM_STEP_TIME;

147
148 Conflict conflict2;
149 conflict2.car = car2Index;
150 conflict2.withCar = car1Index;
151 conflict2.time = time * SIM_STEP_TIME;
152 conflict2.position = node.paths[car2Index][time];
153
154 if (conflicts.find(situation1) == conflicts.end()) {
155 conflicts[situation1] = new std::unordered_set<Conflict>();

170void ManagerOCBS::pathfinding(Node *node, int carIndex) {
171 AStar::node start;
172 start.point = starts[carIndex];
173 start.speed = 0;
174 AStar::node end;
175 end.point = ends[carIndex];
176 end.speed = 0;
177
178 std::unordered_map<AStar::node, AStar::node> cameFrom;
179 std::unordered_map<AStar::node, double> gScore;
180 std::unordered_map<AStar::node, double> fScore;
181
182 auto heuristic = [&](const AStar::node &a) {
183 sf::Vector2f diff = end.point.position - a.point.position;
184 double distance = std::sqrt(diff.x * diff.x + diff.y * diff.y);
185 return distance / CAR_MAX_SPEED_MS;
186 };
187 auto compare = [&](const AStar::node &a, const AStar::node &b) { return fScore[a] > fScore[b]; };
188
189 std::priority_queue<AStar::node, std::vector<AStar::node>, decltype(compare)> openSetAstar(compare);
190 std::unordered_set<AStar::node> isInOpenSet;
191
192 openSetAstar.push(start);
193 gScore[start] = 0;
194 fScore[start] = heuristic(start);
195
196 auto neighbors = graph.getNeighbors();
197
198 int nbIterations = 0;
199 while (!openSetAstar.empty() && nbIterations++ < ASTAR_MAX_ITERATIONS) {
200 AStar::node current = openSetAstar.top();
201 openSetAstar.pop();
202 isInOpenSet.erase(current);
203
204 if (current.point == end.point) {
205 AStar::node currentCopy = current;
206 std::vector<AStar::node> nodePaths;
207
208 while (!(currentCopy == start)) {
209 nodePaths.push_back(currentCopy);
210 currentCopy = cameFrom[currentCopy];
211 }
212
213 nodePaths.push_back(currentCopy);
214 std::reverse(nodePaths.begin(), nodePaths.end());
215
216 double oldCost = node->costs[carIndex];
217 cars[carIndex].assignPath(nodePaths, graph);
218
219 node->paths[carIndex] = cars[carIndex].getPath();
220 node->costs[carIndex] = cars[carIndex].getPathTime();
221 node->cost += node->costs[carIndex] - oldCost;
222
223 spdlog::debug("Found path for car {} with cost: {}", carIndex, node->costs[carIndex]);
224 return;
225 }
226
227 for (const auto &neighborGraphPoint : neighbors[current.point]) {
228 if (current.speed > neighborGraphPoint.maxSpeed)
229 continue;
230
231 if (!neighborGraphPoint.isRightWay && ROAD_ENABLE_RIGHT_HAND_TRAFFIC)
232 continue;

235 newSpeeds.push_back(current.speed);
236
237 double distance = graph.getInterpolator(current.point, neighborGraphPoint)->getDistance();

239 double nSpeedDec = std::sqrt(std::pow(current.speed, 2) - 2 * CAR_DECELERATION * distance);
240

242 int numSpeedDiv = NUM_SPEED_DIVISIONS;
243 for (int i = 1; i < numSpeedDiv + 1; i++) {
244 double s = (current.speed + (nSpeed - current.speed) * i / numSpeedDiv);
245 if (s < SPEED_RESOLUTION)
246 continue;
247 newSpeeds.push_back(s);
248 }
249 };

254 push(nSpeedAcc);
255 }
256
257 if (nSpeedDec == nSpeedDec && std::isfinite(nSpeedDec)) { // check if nSpeedDec is finite and not NaN
258 if (nSpeedDec < 0 && current.speed > 0) {
259 push(0);
260 } else if (nSpeedDec >= 0) {
261 push(nSpeedDec);
262 }
263 }
264
265 AStar::node neighbor;
266 neighbor.point = neighborGraphPoint.point;
267 neighbor.arcFrom = {current.point, neighborGraphPoint};
268 if (distance == 0) {
269 neighbor.speed = current.speed;
270 if (gScore.find(neighbor) == gScore.end() || gScore[current] < gScore[neighbor]) {
271 cameFrom[neighbor] = current;
272 gScore[neighbor] = gScore[current];
273 fScore[neighbor] = gScore[neighbor] + heuristic(neighbor);
274
275 if (isInOpenSet.find(neighbor) == isInOpenSet.end()) {
276 openSetAstar.push(neighbor);
277 isInOpenSet.insert(neighbor);
278 }
279 }
280 continue;
281 }
282
283 for (const auto &newSpeed : newSpeeds) {
284 if (newSpeed > CAR_MAX_SPEED_MS || newSpeed > neighborGraphPoint.maxSpeed || newSpeed < 0)
285 continue;
286
287 if (newSpeed == current.speed && newSpeed == 0)
288 continue;
289
290 neighbor.speed = newSpeed;
291
292 double duration = 2 * distance / (current.speed + newSpeed);
293 double tentativeGScore = gScore[current] + duration;
294
295 double t = gScore[current];
296 bool conflictFree = true;
297
298 // Checking for conflicts
299 DubinsInterpolator *interpolator = graph.getInterpolator(current.point, neighborGraphPoint);
300 for (double tt = 0; tt < duration; tt = tt + SIM_STEP_TIME) {
301 ConflictSituation confS;
302 confS.car = carIndex;
303 confS.at = interpolator->get(tt, current.speed, newSpeed).position;
304 confS.time = t + tt;
305
306 if (conflicts.find(confS) == conflicts.end()) {
307 continue;
308 }

311
312 if (conflictSet->size() == 0) {
313 continue;
314 }
315
316 for (const auto &conf : *conflictSet) {
317 // Check during all the duration if there is a conflict
318 sf::Vector2f diff = confS.at - conf.position;
319 double len = std::sqrt(diff.x * diff.x + diff.y * diff.y);
320
321 if (len < CAR_LENGTH * COLLISION_SAFETY_FACTOR) {
322 conflictFree = false;
323 break;
324 }
325 }
326 if (!conflictFree)
327 break;
328 }
329
330 if (!conflictFree)
331 continue;
332
333 if (gScore.find(neighbor) == gScore.end() || tentativeGScore < gScore[neighbor]) {
334 cameFrom[neighbor] = current;
335 gScore[neighbor] = tentativeGScore;
336 fScore[neighbor] = gScore[neighbor] + heuristic(neighbor);
337
338 if (isInOpenSet.find(neighbor) == isInOpenSet.end()) {
339 openSetAstar.push(neighbor);
340 isInOpenSet.insert(neighbor);
341 }
342 }
343 }
344 }
345 }
