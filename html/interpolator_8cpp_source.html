18void DubinsInterpolator::init(CityGraph::point start_, CityGraph::point end_, double radius_) {
19 startPoint = start_;
20 endPoint = end_;
21 radius = radius_;
22
23 // Create a Dubins state space with the given turning radius
24 // The second parameter (true) indicates symmetric Dubins paths
25 ob::DubinsStateSpace space = ob::DubinsStateSpace(radius, true);
26
27 // Allocate OMPL states for start and end poses
28 ob::State *start = space.allocState();
29 ob::State *end = space.allocState();
30
31 // Set start and end poses (position + orientation)
32 start->as<ob::DubinsStateSpace::StateType>()->setXY(startPoint.position.x, startPoint.position.y);
33 start->as<ob::DubinsStateSpace::StateType>()->setYaw(startPoint.angle.asRadians());
34
35 end->as<ob::DubinsStateSpace::StateType>()->setXY(endPoint.position.x, endPoint.position.y);
36 end->as<ob::DubinsStateSpace::StateType>()->setYaw(endPoint.angle.asRadians());
37
38 // Compute the Dubins path distance
39 distance = space.distance(start, end);
40
41 // Validate the computed distance against straight-line distance
42 sf::Vector2 diff = startPoint.position - endPoint.position;
43 double absDist = std::sqrt(std::pow(diff.x, 2) + std::pow(diff.y, 2));
44
45 // Distance should be at most straight-line distance plus maximum arc length
46 if (distance > absDist + 2 * M_PI * radius) {
47 spdlog::warn("Distance is way too big in DubinsInterpolator");
48 distance = absDist;
49 }
50
51 // Distance should be at least the straight-line distance (with small tolerance)
52 constexpr double DISTANCE_TOLERANCE = 0.1;
53 if (distance + DISTANCE_TOLERANCE < absDist) {
54 spdlog::warn("Distance is way too small in DubinsInterpolator");
55 distance = absDist;
56 }
57
58 // Compute interpolation step size in [0,1] parameter space
59 double dx = DUBINS_INTERPOLATION_STEP / distance;
60 interpolatedCurve.clear();
61 interpolatedCurve.push_back(startPoint);
62
63 // Interpolate points along the Dubins curve
64 for (double x = dx; x < 1; x += dx) {
65 if (x == 1) // Skip endpoint to avoid duplication
66 continue;
67
68 ob::State *state = space.allocState();
69 space.interpolate(start, end, x, state);
70
71 // Extract pose from interpolated state
72 double x_ = state->as<ob::DubinsStateSpace::StateType>()->getX();
73 double y_ = state->as<ob::DubinsStateSpace::StateType>()->getY();
74 double yaw_ = state->as<ob::DubinsStateSpace::StateType>()->getYaw();
75
76 CityGraph::point point;
77 point.position = {(float)x_, (float)y_};
78 point.angle = sf::radians(yaw_);
79
80 interpolatedCurve.push_back(point);
81
82 space.freeState(state);
83 }
84
85 // Add endpoint explicitly
86 interpolatedCurve.push_back(endPoint);
87
88 numInterpolatedPoints = interpolatedCurve.size();
89
90 space.freeState(start);
91 space.freeState(end);

94CityGraph::point DubinsInterpolator::get(double time, double startSpeed, double endSpeed) {
95 // Calculate acceleration based on start/end speeds and path distance
96 // Using kinematic equation: v^2 = u^2 + 2as
97 double acc = (std::pow(endSpeed, 2) - std::pow(startSpeed, 2)) / (2 * distance);
98
99 // Define position function using kinematic equation: s = ut + 0.5at^2
100 // Normalized to [0,1] by dividing by total distance
101 auto xFun = [&](double t) { return (0.5 * acc * std::pow(t, 2) + startSpeed * t) / distance; };
102
103 // Map normalized position to interpolated curve index
104 int index = std::round((numInterpolatedPoints - 1) * xFun(time));
105 index = std::clamp(index, 0, numInterpolatedPoints - 1);
106
107 return interpolatedCurve[index];
108}
